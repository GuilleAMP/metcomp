#include <iostream>
using namespace::std;

int main() {
    cout << "Cuadrados del 1 al 5:\n";

    for (int i = 1; i <= 5; i++) {
        int cuadrado = i * i;
        cout << cuadrado;
	cout << "\n";
    }

    cout << "\nAhora con while:\n";

    int j = 1;
    while (j <= 5) {
        int cuadrado = j * j;
	cout << cuadrado;
	cout << "\n";
        j = j + 1;
    }

    return 0;
}
